\chapter{Objetivos}
% ------------------------------------------------------------------------
% ------------------------------------------------------------------------

\section*{Objetivo general}

\begin{itemize}
   \item[ ] Definir e implementar mecanismos para el despliegue de una infraestructura cloud que provea alta disponibilidad y escalabilidad para el caso de una plataforma iot en un entorno distribuido.
\end{itemize}

\section*{Objetivos específicos}

\begin{itemize}
    \item[] Identificar las herramientas en el mercado para el despliegue de una infraestructura cloud.
	\item[] Diseño de la arquitectura y pruebas para validar la arquitectura planteada.
	\item[] Despliegue de la arquitectura planteada en un entorno de pruebas y aplicación de las pruebas planteadas.
\end{itemize}
